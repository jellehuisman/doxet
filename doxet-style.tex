% simple ConTeXt style sheet for doxet

% begin filter pandoc #

\usemodule[filter]

\defineexternalfilter
    [markdown]
    [filtercommand={pandoc -t context -o \externalfilteroutputfile\space \externalfilterinputfile}]

% SEE: https://github.com/adityam/filter/

% end pandoc filter #

\mainlanguage[en]

\setuppapersize[A4][A4]

\setuplayout[backspace=30mm,width=155mm,topspace=15mm,header=10mm,   footer=10mm,height=260mm]

\setuppagenumbering[location={footer,right}]

\definefontfeature[default][default][mode=node,expansion=quality,protrusion=pure]
\setupalign[hanging,hz]

\setupbodyfontenvironment[default][em=italic]

% serif: Noto Serif, Kreon, Gentium Basic, Droid Serif
% sans: Open Sans, Lato
% mono: Inconsolata

\def\myserif{Gentium Basic}
\def\mysans{Open Sans}
\def\mymono{Inconsolata}
\def\myccicons{CC Icons}

\definefontfamily[mainface][serif][\myserif]
\definefontfamily[mainface][sans][\mysans]
\definefontfamily[mainface][mono][\mymono]
\definefontfamily[mainface][handwriting][\myccicons]

\setupbodyfont[mainface,12pt]

\setupinteraction[state=start,color=darkblue,contrastcolor=darkblue]

\def\execsum#1{{\bTABLE[frame=off,background=color,backgroundcolor=lightgray] \bTR \bTD {\ss {#1}} \eTD \eTR \eTABLE\par \white{.}}}

% use sans headers

\setuphead [chapter][style=\ssd,color=darkred]%dark,middle,light
\setuphead [section][style=\ssc,color=middleblue]
\setuphead [subsection][style=\ssb,color=middleblue]%lightgreen]
\setuphead [subsubsection][style=\ssa,color=middleblue]%darkgray]

\setuphead [title][style=\ssd,color=middlered]%dark,middle,light
\setuphead [subject][style=\ssc,color=middleblue]
\setuphead [subsubject][style=\ssb,color=middleblue]%lightgreen]
\setuphead [subsubsubject][style=\ssa,color=middleblue]%darkgray]
